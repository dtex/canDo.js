{ events }

canDoConfiguration.events = { [ 'event': callBackFunction ] }

Description: Binds an event to the Canvas element.

{
	string event: function callBackFunction()
}

Example

var clickable = CanDo('translate', {
   events: {
    	click: function() {alert('I\'ve been clicked');}
    },
    play: function() { drawMyButton(); }
});
